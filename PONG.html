    <title>Juego de Pong</title>

    <style>

        body {

            display: flex;

            justify-content: center;

            align-items: center;

            height: 100vh;

            margin: 0;

            background-color: #000;

        }

        canvas {

            border: 2px solid #fff;

        }

    </style>

<body>

    <canvas id="pongCanvas" width="800" height="400"></canvas>

    <script>

        const canvas = document.getElementById('pongCanvas');

        const context = canvas.getContext('2d');


        const paddleWidth = 10, paddleHeight = 100;

        const ballSize = 10;

        let playerY = (canvas.height - paddleHeight) / 2;

        let cpuY = (canvas.height - paddleHeight) / 2;

        let ballX = canvas.width / 2, ballY = canvas.height / 2;

        let ballSpeedX = 5, ballSpeedY = 5;

        const playerSpeed = 8;

        let playerScore = 0, cpuScore = 0;

        const winningScore = 5;

        let gameOver = false;


        function drawRect(x, y, width, height, color) {

            context.fillStyle = color;

            context.fillRect(x, y, width, height);

        }


        function drawCircle(x, y, radius, color) {

            context.fillStyle = color;

            context.beginPath();

            context.arc(x, y, radius, 0, Math.PI * 2, false);

            context.closePath();

            context.fill();

        }


        function drawNet() {

            for (let i = 0; i < canvas.height; i += 20) {

                drawRect(canvas.width / 2 - 1, i, 2, 10, '#fff');

            }

        }


        function resetBall() {

            if (playerScore >= winningScore || cpuScore >= winningScore) {

                gameOver = true;

            }

            ballX = canvas.width / 2;

            ballY = canvas.height / 2;

            ballSpeedX = -ballSpeedX;

        }


        function moveBall() {

            if (gameOver) return;


            ballX += ballSpeedX;

            ballY += ballSpeedY;


            if (ballY <= 0 || ballY >= canvas.height) {

                ballSpeedY = -ballSpeedY;

            }


            if (ballX <= paddleWidth) {

                if (ballY > playerY && ballY < playerY + paddleHeight) {

                    ballSpeedX = -ballSpeedX;

                } else if (ballX < 0) {

                    cpuScore++;

                    resetBall();

                }

            }


            if (ballX >= canvas.width - paddleWidth) {

                if (ballY > cpuY && ballY < cpuY + paddleHeight) {

                    ballSpeedX = -ballSpeedX;

                } else if (ballX > canvas.width) {

                    playerScore++;

                    resetBall();

                }

            }

        }


        function movePaddles() {

            if (ballY > cpuY + paddleHeight / 2) {

                cpuY += playerSpeed;

            } else {

                cpuY -= playerSpeed;

            }

        }


        function update() {

            moveBall();

            movePaddles();

        }


        function render() {

            drawRect(0, 0, canvas.width, canvas.height, '#000');

            drawNet();

            drawRect(0, playerY, paddleWidth, paddleHeight, '#fff');

            drawRect(canvas.width - paddleWidth, cpuY, paddleWidth, paddleHeight, '#fff');

            drawCircle(ballX, ballY, ballSize, '#fff');


            if (gameOver) {

                context.fillStyle = '#fff';

                context.font = '48px Arial';

                const message = playerScore >= winningScore ? '¡Ganaste!' : 'La CPU ganó';

                context.fillText(message, canvas.width / 2 - context.measureText(message).width / 2, canvas.height / 2);

                context.font = '24px Arial';

                context.fillText('Haz clic para reiniciar', canvas.width / 2 - context.measureText('Haz clic para reiniciar').width / 2, canvas.height / 2 + 40);

            }

        }


        function gameLoop() {

            update();

            render();

        }


        setInterval(gameLoop, 1000 / 60);


        window.addEventListener('mousemove', (event) => {

            const rect = canvas.getBoundingClientRect();

            const root = document.documentElement;

            const mouseY = event.clientY - rect.top - root.scrollTop;

            playerY = mouseY - paddleHeight / 2;

        });


        canvas.addEventListener('click', () => {

            if (gameOver) {

                playerScore = 0;

                cpuScore = 0;

                gameOver = false;

            }

        });

    </script>

</body>
